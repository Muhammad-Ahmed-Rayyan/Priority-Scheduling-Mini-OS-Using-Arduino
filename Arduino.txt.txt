#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// ---------- LCD ----------
LiquidCrystal_I2C lcd(0x27, 16, 2);

// ---------- Pins ----------
const int greenLED = 7;
const int yellowLED = 8;
const int redLED = 10;
const int buzzer = 9;
const int potPin = A0;
const int buttonPin = 2;

// ---------- Variables ----------
volatile bool manualReset = false;

// ---------- Priority Thresholds ----------
const int LOW_MAX = 341;
const int MED_MAX = 682;

// ---------- Setup ----------
void setup() {
  pinMode(greenLED, OUTPUT);
  pinMode(yellowLED, OUTPUT);
  pinMode(redLED, OUTPUT);
  pinMode(buzzer, OUTPUT);
  pinMode(buttonPin, INPUT_PULLUP);

  lcd.init();
  lcd.backlight();

  Serial.begin(9600);

  // ---------- BOOT SEQUENCE ----------
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Starting....");
  lcd.setCursor(0, 1);
  lcd.print("OS Initializing");
  delay(1500);

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("PRIORITY OS");
  lcd.setCursor(0, 1);
  lcd.print("Ready");
  delay(2000);

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Rayyan & Namish");
  lcd.setCursor(0, 1);
  lcd.print("   OS Project");
  delay(4000);

  lcd.clear();

  attachInterrupt(digitalPinToInterrupt(buttonPin), manualResetISR, FALLING);
}

// ---------- Loop ----------
void loop() {

  // ---------- MANUAL RESET ----------
  if (manualReset) {
    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("MANUAL RESET");
    lcd.setCursor(0, 1);
    lcd.print("System Reset");

    digitalWrite(redLED, HIGH);
    digitalWrite(buzzer, HIGH);
    delay(1000);

    digitalWrite(redLED, LOW);
    digitalWrite(buzzer, LOW);

    manualReset = false;
    return;
  }

  // ---------- Read Potentiometer ----------
  int potValue = readPotSmooth();

  Serial.print("Pot: ");
  Serial.println(potValue);

  // ---------- Reset Outputs ----------
  digitalWrite(greenLED, LOW);
  digitalWrite(yellowLED, LOW);
  digitalWrite(redLED, LOW);
  digitalWrite(buzzer, LOW);

  String task;

  // ---------- Priority Scheduling ----------
  if (potValue > MED_MAX) {
    task = "HIGH PRIORTY";
    digitalWrite(redLED, HIGH);
    digitalWrite(buzzer, HIGH);
  }
  else if (potValue > LOW_MAX) {
    task = "MEDIUM PRIORTY";
    digitalWrite(yellowLED, HIGH);
  }
  else {
    task = "LOW PRIORTY";
    digitalWrite(greenLED, HIGH);
  }

  // ---------- LCD UPDATE ----------
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Task:");
  lcd.print(task);

  lcd.setCursor(0, 1);
  lcd.print("Input: ");
  lcd.print(potValue);

  delay(150);
}

// ---------- POTENTIOMETER SMOOTHING ----------
int readPotSmooth() {
  long sum = 0;
  for (int i = 0; i < 10; i++) {
    sum += analogRead(potPin);
    delay(2);
  }
  return sum / 10;
}

// ---------- INTERRUPT ----------
void manualResetISR() {
  manualReset = true;
}
